// --- Shared Helper ---
const formatCurrency = (num) => {
  if (num === undefined || num === null || num === '') return '0.00';
  const cleanNum = typeof num === 'string' ? num.replace(/,/g, '') : num;
  const n = Number(cleanNum);
  return isNaN(n) ? '0.00' : n.toLocaleString('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
};

// --- Shared Pagination Hook ---
const usePagination = (data) => {
  const [currentPage, setCurrentPage] = useState(1);
  const [pageSize, setPageSize] = useState(10);

  useEffect(() => {
    setCurrentPage(1);
  }, [data.length, pageSize]);

  const totalPages = Math.ceil(data.length / pageSize) || 1;
  const currentData = data.slice((currentPage - 1) * pageSize, currentPage * pageSize);

  return { currentPage, setCurrentPage, pageSize, setPageSize, totalPages, currentData, totalCount: data.length };
};

const PaginationFooter = ({ totalCount, currentPage, setCurrentPage, totalPages }) => {
  const [jumpPage, setJumpPage] = useState('');

  const handleJump = (e) => {
    if (e.key === 'Enter') {
      const page = parseInt(jumpPage, 10);
      if (!isNaN(page) && page >= 1 && page <= totalPages) {
        setCurrentPage(page);
      }
      setJumpPage('');
    }
  };

  const renderPageNumbers = () => {
    const pages = [];
    let start = Math.max(1, currentPage - 2);
    let end = Math.min(totalPages, start + 4);
    if (end - start < 4) {
      start = Math.max(1, end - 4);
    }
    
    for (let i = start; i <= end; i++) {
      pages.push(
        <button
          key={i}
          onClick={() => setCurrentPage(i)}
          className={`w-8 h-8 rounded-md text-sm font-medium transition-colors ${currentPage === i ? 'bg-blue-500 text-white shadow-sm' : 'text-gray-600 hover:bg-gray-100'}`}
        >
          {i}
        </button>
      );
    }
    return pages;
  };

  return (
    <div className="flex items-center justify-between px-6 py-4 border-t border-gray-100 bg-white shrink-0">
      <div className="text-sm text-gray-500">共 {totalCount} 條</div>
      <div className="flex items-center gap-6">
        <div className="flex items-center gap-1">
          <button onClick={() => setCurrentPage(p => Math.max(1, p - 1))} disabled={currentPage === 1} className="w-8 h-8 flex items-center justify-center rounded-md text-gray-400 hover:bg-gray-100 disabled:opacity-50 transition-colors"><ChevronRight size={16} className="rotate-180" /></button>
          {renderPageNumbers()}
          <button onClick={() => setCurrentPage(p => Math.min(totalPages, p + 1))} disabled={currentPage === totalPages} className="w-8 h-8 flex items-center justify-center rounded-md text-gray-400 hover:bg-gray-100 disabled:opacity-50 transition-colors"><ChevronRight size={16} /></button>
        </div>
        <div className="flex items-center gap-2 text-sm text-gray-600">
          前往
          <input
            type="text"
            value={jumpPage}
            onChange={(e) => setJumpPage(e.target.value)}
            onKeyDown={handleJump}
            className="w-12 text-center border border-gray-200 bg-gray-50/50 rounded-lg py-1 outline-none focus:bg-white focus:border-blue-500 transition-all"
          />
          頁
        </div>
      </div>
    </div>
  );
};

    const filteredRecords = inventoryRecords.filter(r => {
      if (r.itemId !== item.id) return false;
      if (appliedStartDate && r.date < appliedStartDate) return false;
      if (appliedEndDate && r.date > appliedEndDate + ' 23:59:59') return false;
      if (appliedReason && !r.reason.includes(appliedReason)) return false;
      return true;
    });

    const paginationProps = usePagination(filteredRecords);

    const handleSearch = () => {
      setAppliedStartDate(startDate);
      setAppliedEndDate(endDate);
      setAppliedReason(reason);
    };

    const handleReset = () => {
      setStartDate('');
      setEndDate('');
      setReason('');
      setAppliedStartDate('');
      setAppliedEndDate('');
      setAppliedReason('');
    };

    return (
      <div className="fixed inset-0 bg-black/20 backdrop-blur-sm flex items-center justify-center z-50 animate-in fade-in duration-200">
        <div className="bg-white rounded-[24px] shadow-[0_20px_50px_rgba(0,0,0,0.1)] w-[1000px] max-h-[90vh] flex flex-col overflow-hidden border border-white/40">
          <div className="flex justify-between items-center px-8 py-5 border-b border-gray-100 bg-gray-50/50 backdrop-blur-md">
            <h2 className="text-lg font-bold text-gray-900 tracking-tight">庫存變動記錄</h2>
            <button onClick={closeModal} className="text-gray-400 hover:bg-gray-200/50 hover:text-gray-700 p-2 rounded-full transition-all active:scale-95"><X size={20}/></button>
          </div>
          
          <div className="p-8 overflow-y-auto flex-1 bg-[#F5F5F7]">
            <div className="bg-white p-6 rounded-2xl shadow-[0_2px_10px_rgba(0,0,0,0.02)] border border-gray-100 mb-6 flex flex-wrap gap-4 items-center">
              <div className="flex items-center border border-gray-200 bg-gray-50/50 rounded-xl overflow-hidden shadow-sm text-sm">
                <span className="text-gray-500 bg-gray-100/50 px-3 py-1.5 border-r border-gray-200/50 whitespace-nowrap">變動時間</span>
                <input type="date" value={startDate} onChange={(e) => setStartDate(e.target.value)} className="p-1.5 outline-none w-36 bg-transparent text-gray-700 px-3" />
                <span className="px-2 text-gray-400">-</span>
                <input type="date" value={endDate} onChange={(e) => setEndDate(e.target.value)} className="p-1.5 outline-none w-36 bg-transparent text-gray-700 px-3" />
              </div>
              <div className="w-56"><AppleInput label="變動原因" placeholder="請輸入" defaultValue={reason} onChange={(e) => setReason(e.target.value)} /></div>
              <div className="flex-1"></div>
              <div className="flex gap-2">
                <AppleButtonSecondary onClick={handleReset}>重置</AppleButtonSecondary>
                <AppleButtonPrimary onClick={handleSearch}>查詢</AppleButtonPrimary>
              </div>
            </div>

            <div className="bg-gray-100/80 p-4 rounded-xl border border-gray-200 mb-6 flex flex-wrap gap-x-8 gap-y-3 text-sm">
              <div className="flex items-center"><span className="text-gray-500 mr-2">服務編號：</span><span className="font-mono text-gray-900">{item.id}</span></div>
              <div className="flex items-center"><span className="text-gray-500 mr-2">服務名稱：</span><span className="font-medium text-gray-900">{item.name}</span></div>
              <div className="flex items-center"><span className="text-gray-500 mr-2">現有庫存：</span><span className={`font-medium text-blue-600 ${item.stock === '無限' ? 'text-2xl leading-none' : 'text-base'}`}>{item.stock === '無限' ? '∞' : item.stock}</span></div>
            </div>

            <div className="mb-4 flex justify-end">
              <PageSizeSelector pageSize={paginationProps.pageSize} setPageSize={paginationProps.setPageSize} />
            </div>

            <div className="bg-white rounded-2xl shadow-[0_2px_10px_rgba(0,0,0,0.02)] border border-gray-100 flex flex-col">
              <div className="overflow-x-auto">
                <table className="w-full text-sm text-left min-w-[800px]">
                  <thead className="bg-gray-50/80 text-gray-500 border-b border-gray-100">
                    <tr>
                      <th className="py-3 px-4 font-medium whitespace-nowrap">變更時間</th>
                      <th className="py-3 px-4 font-medium whitespace-nowrap">變更編號</th>
                      <th className="py-3 px-4 font-medium whitespace-nowrap">服務類型</th>
                      <th className="py-3 px-4 font-medium whitespace-nowrap">變更原因</th>
                      <th className="py-3 px-4 font-medium whitespace-nowrap">變更數量</th>
                      <th className="py-3 px-4 font-medium text-center whitespace-nowrap">操作</th>
                    </tr>
                  </thead>
                  <tbody className="divide-y divide-gray-50 text-gray-700">
                    {paginationProps.currentData.length === 0 ? (
                      <tr>
                        <td colSpan="6" className="py-8 text-center text-gray-500">暫無變動記錄</td>
                      </tr>
                    ) : (
                      paginationProps.currentData.map((r, i) => (
                        <tr key={i} className="hover:bg-gray-50/50 transition-colors">
                          <td className="py-4 px-4 whitespace-nowrap">{r.date}</td>
                          <td className="py-4 px-4 text-xs font-mono text-gray-500">{r.id}</td>
                          <td className="py-4 px-4">{r.type}</td>
                          <td className="py-4 px-4">{r.reason}</td>
                          <td className="py-4 px-4">
                            <span className={`font-medium ${r.qty > 0 ? 'text-blue-600' : 'text-orange-500'}`}>{r.qty > 0 ? `+${r.qty}` : r.qty}</span>
                          </td>
                          <td className="py-4 px-4 text-center">
                            {/* 依影片展示，操作欄暫留空 */}
                          </td>
                        </tr>
                      ))
                    )}
                  </tbody>
                </table>
              </div>
              <PaginationFooter {...paginationProps} />
            </div>
          </div>
          
          <div className="p-6 border-t border-gray-100 bg-white flex justify-end">
            <AppleButtonSecondary onClick={closeModal}>關閉</AppleButtonSecondary>
          </div>
        </div>
      </div>
    );
  };

  const ModalBillDetails = () => {
    const bill = selectedCustomer;
    if (modalType !== 'billDetails' || !bill) return null;

    const monthRecords = serviceRecords.filter(r => r.customerId === bill.customerId && r.date.startsWith(bill.month));
    const servicesTotal = monthRecords.reduce((sum, r) => sum + (parseFloat(String(r.amount || 0).replace(/,/g, '')) || 0), 0);
    const basicFee = parseFloat(String(bill.basicFee || 9900).replace(/,/g, '')) || 0;
    const prepaid = parseFloat(String(bill.prepaid || 0).replace(/,/g, '')) || 0;
    const totalFee = basicFee + servicesTotal;
    const amountPayable = totalFee - prepaid;

    return (
      <div className="fixed inset-0 bg-black/20 backdrop-blur-sm flex items-center justify-center z-50 animate-in fade-in duration-200">
        <div className="bg-white rounded-[24px] shadow-[0_20px_50px_rgba(0,0,0,0.1)] w-[700px] flex flex-col overflow-hidden border border-white/40">
          <div className="flex justify-between items-center px-8 py-5 border-b border-gray-100 bg-gray-50/50 backdrop-blur-md">
            <h2 className="text-lg font-bold text-gray-900 tracking-tight">結算詳情</h2>
            <button onClick={closeModal} className="text-gray-400 hover:bg-gray-200/50 hover:text-gray-700 p-2 rounded-full transition-all active:scale-95"><X size={20}/></button>
          </div>
          
          <div className="p-8">
            <div className="flex items-center justify-between text-sm text-gray-600 mb-6 bg-gray-50/80 p-4 rounded-xl border border-gray-100">
              <div className="flex gap-8">
                <span>住院號: <span className="font-medium text-gray-900">{bill.customerId}</span></span>
                <span>客戶姓名: <span className="font-medium text-gray-900">{bill.name}</span></span>
              </div>
              <span>賬單月份: <span className="font-medium text-gray-900">{bill.month}</span></span>
            </div>

            <div className="border border-gray-200 rounded-xl overflow-hidden shadow-sm">
              <table className="w-full text-sm text-left">
                <thead className="bg-gray-50/80 text-gray-500 border-b border-gray-200">
                  <tr>
                    <th className="py-3 px-4 font-medium">收費項目</th>
                    <th className="py-3 px-4 font-medium">服務名稱</th>
                    <th className="py-3 px-4 font-medium text-right">金額(MOP)</th>
                  </tr>
                </thead>
                <tbody className="divide-y divide-gray-100 text-gray-700">
                  <tr>
                    <td className="py-4 px-4 font-medium text-gray-900">基本院費 <span className="text-xs text-gray-400 font-normal ml-1">(交下月院費)</span></td>
                    <td className="py-4 px-4"></td>
                    <td className="py-4 px-4 text-right font-medium">{formatCurrency(basicFee)}</td>
                  </tr>
                  {monthRecords.map((r, i) => (
                    <tr key={i}>
                      <td className="py-4 px-4 text-gray-500">服務收費</td>
                      <td className="py-4 px-4">{r.name}</td>
                      <td className="py-4 px-4 text-right">{formatCurrency(r.amount)}</td>
                    </tr>
                  ))}
                  <tr className="bg-gray-50/30">
                    <td className="py-4 px-4 font-bold text-gray-900">費用合計</td>
                    <td className="py-4 px-4"></td>
                    <td className="py-4 px-4 text-right font-bold text-gray-900">{formatCurrency(totalFee)}</td>
                  </tr>
                  <tr>
                    <td className="py-4 px-4 font-medium text-gray-900">預繳費用</td>
                    <td className="py-4 px-4"></td>
                    <td className="py-4 px-4 text-right text-blue-600 font-medium">{prepaid > 0 ? `-${formatCurrency(prepaid)}` : '0.00'}</td>
                  </tr>
                  <tr className="bg-blue-50/30">
                    <td className="py-4 px-4 font-bold text-gray-900">應繳費用</td>
                    <td className="py-4 px-4"></td>
                    <td className="py-4 px-4 text-right font-bold text-red-600 text-base">{formatCurrency(amountPayable)}</td>
                  </tr>
                </tbody>
              </table>
            </div>
          </div>
        </div>
      </div>
    );
  };

  return (
    <div className="flex flex-col h-screen font-sans bg-[#F5F5F7] overflow-hidden text-gray-800 selection:bg-blue-100 selection:text-blue-900">
      <TopBar />
      <div className="flex flex-1 overflow-hidden relative">
        <Sidebar />
        
        {activeMenu === '客戶信息管理' && <CustomerView />}
        {activeMenu.startsWith('服務記錄') && <ServiceRecordView customerId={activeMenu.split('-')[1]} />}
        {activeMenu === '庫存概覽' && <InventoryView />}
        {activeMenu === '服務管理' && <ServicesView />}
        {activeMenu === '賬單管理' && <BillsView />}
        {activeMenu === '角色管理' && <RoleManagementView />}
        {activeMenu === '新增角色' && <RoleAddView />}
        {activeMenu === '編輯角色' && <RoleEditView />}
        {activeMenu === '用戶管理' && <UserManagementView />}
        {activeMenu === '編輯用戶' && <UserEditView />}
      </div>

      {modalType === 'details' && <ModalDetails />}
      {(modalType === 'edit' || modalType === 'add') && <ModalEdit />}
      {(modalType === 'addService' || modalType === 'editService') && <ModalService />}
      <ModalServiceRecordAdd />
      <ModalServiceRecordEdit />
      <ModalInventoryChangeSingle />
      <ModalInventoryChangeBulk />
      <ModalInventoryRecord />
      <ModalBillDetails />
      <ModalConfirmDelete />
      <ModalNotAvailable />
    </div>
  );
}
